const express = require('express'); const cors = require('cors'); require('dotenv').config(); const { createClient } = require('@supabase/supabase-js'); const app = express(); const PORT = process.env.PORT || 8080; // ─── Supabase client ───────────────────────────────────────────── const supabase = createClient( process.env.SUPABASE_URL, process.env.SUPABASE_ANON_KEY ); // ─── Middleware ─────────────────────────────────────────────────── app.use(cors({ origin: '*', // tighten this once you have a domain methods: ['GET', 'POST'], })); app.use(express.json()); // ─── Waitlist / email capture ──────────────────────────────── app.post('/waitlist', async (req, res) => { try { const { email } = req.body; if (!email || !email.includes('@')) { return res.status(400).json({ error: 'valid email required' }); } const { error } = await supabase .from('waitlist') .insert({ email, source: 'extension', created_at: new Date().toISOString() }) .select(); if (error && !error.message.includes('duplicate')) { console.error('Waitlist insert error:', error); } res.json({ status: 'ok' }); } catch (e) { res.json({ status: 'ok' }); // Silent fail — never block the user } }); // ─── Health check ───────────────────────────────────────────────── app.get('/', (req, res) => { res.json({ status: 'ok', product: 'Transparé API', version: '1.0.0' }); }); // ─── Normalize ingredient name ──────────────────────────────────── function normalize(name) { return name .toLowerCase() .replace(/[()]/g, '') .replace(/\s+/g, ' ') .trim(); } // ─── Parse ingredient list string into array ────────────────────── // Returns { name, position, total } objects function parseIngredients(raw) { if (!raw) return []; const match = raw.match(/ingredients?[:\s]+(.+)/si); const block = match ? match[1] : raw; const names = block .split(/[,;\/\n]+/) .map(i => normalize(i.replace(/\(.*?\)/g, ''))) .filter(i => i.length > 2 && i.length < 120); const total = names.length; return names.map((name, idx) => ({ name, position: idx + 1, total })); } // ─── Concentration label ─────────────────────────────────────── function getConcentration(position, total) { if (!position || !total) return null; const pct = position / total; if (pct <= 0.25) return 'HIGH CONC.'; if (pct <= 0.60) return 'MID CONC.'; return 'LOW CONC.'; } // ─── Resolve extract base ingredient ───────────────────────── // "moringa oleifera seed extract" -> check "moringa oleifera seed oil" / "moringa oil" // Only flag if the base ingredient is comedogenic async function resolveExtract(ingName) { // Only process if ingredient ends with "extract" if (!ingName.endsWith('extract')) return null; // Strip "extract" and common suffixes to get base name const base = ingName .replace(/\s*(seed|leaf|root|flower|fruit|bark|stem|powder|oil|kernel)?\s*extract$/i, '') .trim(); if (base.length < 4) return null; // Check if base name or its oil/butter form is comedogenic const variants = [ base, base + ' oil', base + ' seed oil', base + ' butter', base + ' kernel oil', ]; for (const variant of variants) { const { data } = await supabase .from('ingredients') .select('id, name, display_name, type, comedogenic_rating, description, confidence, sources') .eq('name', variant) .single(); if (data && data.type === 'comedogenic') { // Return the base ingredient as the flag, with note it was from extract return { ...data, display_name: data.display_name + ' (extract)', description: data.description + ' Extract form also flagged.', }; } // Also check safe list — if base is safe, don't flag if (data && data.type === 'safe') return null; } // Also check aliases for the base name const { data: aliasData } = await supabase .from('ingredient_aliases') .select('ingredient_id, ingredients(id, name, display_name, type, comedogenic_rating, description, confidence, sources)') .eq('alias', base) .single(); if (aliasData?.ingredients?.type === 'comedogenic') { return { ...aliasData.ingredients, display_name: aliasData.ingredients.display_name + ' (extract)', }; } // Unknown extract — don't flag (safe default) return null; } // ─── Score calculator ───────────────────────────────────────────── function calculateScore(flags) { if (flags.length === 0) return 95; let penalty = 0; for (const flag of flags) { if (flag.type === 'comedogenic') { penalty += ((flag.comedogenic_rating || 3) * 6); } else if (flag.type === 'toxic') { penalty += 18; } else if (flag.type === 'irritant') { penalty += 10; } } return Math.max(5, Math.round(100 - penalty)); } // ─── POST /analyze ──────────────────────────────────────────────── // Body: { ingredients: "Water, Glycerin, Isopropyl Myristate...", retailer: "sephora" } // Returns: { score, flags, alternatives, ingredientCount } app.post('/analyze', async (req, res) => { try { const { ingredients: rawIngredients, retailer, productName, productBrand } = req.body; if (!rawIngredients) { return res.status(400).json({ error: 'ingredients field required' }); } const parsed = parseIngredients(rawIngredients); if (parsed.length === 0) { return res.status(400).json({ error: 'could not parse ingredient list' }); } // ── Step 1: Batch lookup — one query for all ingredients ────── const flags = []; const flaggedIds = new Set(); // Build position lookup map: ingredient name -> {position, total} const positionMap = {}; for (const obj of parsed) { positionMap[obj.name] = { position: obj.position, total: obj.total }; } const uniqueNames = [...new Set(parsed.map(p => p.name))]; const extractIngredientObjs = parsed.filter(p => p.name.includes('extract')); const regularIngredients = uniqueNames.filter(n => !n.includes('extract')); const firstTwoWords = regularIngredients.map(n => n.split(' ').slice(0,2).join(' ')).filter(n => n.length > 6); const allLookupTerms = [...new Set([...regularIngredients, ...firstTwoWords])]; // ── Batch 1: exact match ─────────────────────────────────────── const { data: exactMatches } = await supabase .from('ingredients') .select('id, name, display_name, type, comedogenic_rating, description, confidence, sources') .in('name', allLookupTerms) .in('type', ['comedogenic', 'irritant', 'toxic']); if (exactMatches) { for (const match of exactMatches) { if (!flaggedIds.has(match.id)) { const pos = positionMap[match.name] || {}; flags.push({ ...match, position: pos.position || null, total: pos.total || null, concentration: getConcentration(pos.position, pos.total), }); flaggedIds.add(match.id); } } } // ── Batch 2: alias match ─────────────────────────────────────── const { data: aliasMatches } = await supabase .from('ingredient_aliases') .select('alias, ingredients(id, name, display_name, type, comedogenic_rating, description, confidence, sources)') .in('alias', allLookupTerms); if (aliasMatches) { for (const am of aliasMatches) { const ing = am.ingredients; if (ing && ['comedogenic','irritant','toxic'].includes(ing.type) && !flaggedIds.has(ing.id)) { const pos = positionMap[am.alias] || {}; flags.push({ ...ing, position: pos.position || null, total: pos.total || null, concentration: getConcentration(pos.position, pos.total), }); flaggedIds.add(ing.id); } } } // ── Batch 3: extract resolution (parallel) ──────────────────── if (extractIngredientObjs.length > 0) { const extractResults = await Promise.all( extractIngredientObjs.map(obj => resolveExtract(obj.name).then(r => r ? { ...r, position: obj.position, total: obj.total, concentration: getConcentration(obj.position, obj.total), } : null)) ); for (const result of extractResults) { if (result && !flaggedIds.has(result.id)) { flags.push(result); flaggedIds.add(result.id); } } } // ── Step 2: calculate score ──────────────────────────────────── const score = calculateScore(flags); // ── Step 3: get alternatives ─────────────────────────────────── // Determine product category from ingredient patterns const category = guessCategory(parsed, productName); let { data: alternatives } = await supabase .from('alternatives') .select('id, name, brand, clyr_score, price_usd, affiliate_url, retailer, category') .eq('category', category) .order('clyr_score', { ascending: false }) .limit(3); if (!alternatives || alternatives.length === 0) { // Fallback to any high-scoring alternatives let { data: fallbackAlts } = await supabase .from('alternatives') .select('id, name, brand, clyr_score, price_usd, affiliate_url, retailer, category') .order('clyr_score', { ascending: false }) .limit(3); alternatives = fallbackAlts || []; } // ── Step 4: log scan event (anonymized) ─────────────────────── await supabase.from('scan_events').insert({ retailer: retailer || 'unknown', clyr_score: score, flag_count: flags.length, flag_types: [...new Set(flags.map(f => f.type))], }); // ── Step 5: return response ──────────────────────────────────── return res.json({ score, flags: flags.map(f => ({ name: f.display_name || f.name, type: f.type, comedogenic_rating: f.comedogenic_rating, description: f.description, })), alternatives: alternatives.map(a => ({ name: a.name, brand: a.brand, score: a.clyr_score, price: a.price_usd, url: a.affiliate_url, retailer: a.retailer, category: a.category, })), ingredientCount: parsed.length, category, }); } catch (err) { console.error('Analyze error:', err); return res.status(500).json({ error: 'internal server error' }); } }); // ─── GET /ingredients/search?q=isopropyl ───────────────────────── app.get('/ingredients/search', async (req, res) => { const { q } = req.query; if (!q || q.length < 2) return res.status(400).json({ error: 'query too short' }); const { data, error } = await supabase .from('ingredients') .select('name, display_name, type, comedogenic_rating, description') .ilike('name', `%${q.toLowerCase()}%`) .limit(10); if (error) return res.status(500).json({ error: error.message }); return res.json({ results: data }); }); // ─── Category guesser ───────────────────────────────────────────── function guessCategory(ingredients, productName) { const name = (productName || '').toLowerCase(); const ings = ingredients.join(' '); if (name.includes('spf') || name.includes('sunscreen') || ings.includes('titanium dioxide') || ings.includes('zinc oxide')) return 'spf'; if (name.includes('serum')) return 'serum'; if (name.includes('cleanser') || name.includes('wash') || name.includes('foam')) return 'cleanser'; if (name.includes('toner')) return 'toner'; if (name.includes('moisturizer') || name.includes('cream') || name.includes('lotion')) return 'moisturizer'; return 'moisturizer'; // default } // ─── Start server ───────────────────────────────────────────────── app.listen(PORT, () => { console.log(`Transparé API running on port ${PORT}`); });
